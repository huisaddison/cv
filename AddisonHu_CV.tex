\section{Papers}
\cventry{December 2017}{Addison J. Hu and Sahand N. Negahban}{Minimax estimation of bandable
    precision matrices}{Advances in Neural Information Processing Systems}{}{
    - Bandable precision matrices arise naturally
    in a number of time-series contexts 
    and correspond to the adjacency matrix representation of the Gauss-Markov graph.
    Minimax results under the spectral norm have previously been established for
    covariance matrices, both sparse and banded, and for sparse precision matrices.
    We establish minimax bounds for estimating banded precision matrices under the
    spectral norm.
    \\
    - Journal version at \url{https://arxiv.org/abs/1710.07006}
}
\cventry{Summer 2020}{Hu, Kuusela, Lee, Giglio, Wood}{Estimation of tropical-cyclone
    induced ocean thermal flux}{To be submitted}{}{
	- We present a general methodological framework for characterizing the ocean
	thermal response to the passage of tropical cyclones, leveraging oceanographic
	data from Argo profiles and satellite-derived hurricane track observations.
    This is achieved by a multi-step process, which includes characterizing the
    warming of the ocean surface during the hurricane season, as well as
    carefully modelling the variability in the temperature profile differences,
    and culminates in a thin plate spline fit that leverages the aforementioned
    variability estimates to properly account for correlation between the
    observations.  In doing so, we reproduce past scientific results and
    present new insights.
}
